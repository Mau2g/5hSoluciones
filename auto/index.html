enum Combustible { GASOLINA, BIOETANOL, DIESEL, BIODIESEL, GAS_NATURAL }
enum TipoAuto { CIUDAD, SUBCOMPACTO, COMPACTO, FAMILIAR, EJECUTIVO, SUV }
enum Color { BLANCO, NEGRO, ROJO, NARANJA, AMARILLO, VERDE, AZUL, VIOLETA }

class Automovil {
    private String marca;
    private int modelo;
    private double motor;
    private Combustible combustible;
    private TipoAuto tipo;
    private int puertas;
    private int asientos;
    private int velocidadMax;
    private Color color;
    private int velocidadActual;
    private boolean automatico; // ejercicio propuesto
    private int multas; // ejercicio propuesto

    public Automovil(String marca, int modelo, double motor, Combustible combustible, TipoAuto tipo, int puertas, int asientos, int velocidadMax, Color color, boolean automatico) { // ejercicio propuesto
        this.marca = marca;
        this.modelo = modelo;
        this.motor = motor;
        this.combustible = combustible;
        this.tipo = tipo;
        this.puertas = puertas;
        this.asientos = asientos;
        this.velocidadMax = velocidadMax;
        this.color = color;
        this.automatico = automatico; // ejercicio propuesto
        this.velocidadActual = 0;
        this.multas = 0; // ejercicio propuesto
    }

    public boolean isAutomatico() { // ejercicio propuesto
        return automatico;
    }

    public void setAutomatico(boolean automatico) { // ejercicio propuesto
        this.automatico = automatico;
    }

    public void acelerar(int incremento) {
        velocidadActual += incremento;
        if (velocidadActual > velocidadMax) { // ejercicio propuesto
            multas += 500; // ejercicio propuesto
            velocidadActual = velocidadMax; // ejercicio propuesto (se fuerza al límite)
        }
    }

    public void desacelerar(int decremento) {
        velocidadActual -= decremento;
        if (velocidadActual < 0) {
            velocidadActual = 0;
        }
    }

    public void frenar() {
        velocidadActual = 0;
    }

    public double tiempoLlegada(double distancia) {
        if (velocidadActual == 0) return Double.POSITIVE_INFINITY;
        return distancia / velocidadActual;
    }

    public boolean tieneMultas() { // ejercicio propuesto
        return multas > 0;
    }

    public int valorMultas() { // ejercicio propuesto
        return multas;
    }

    public void mostrarDatos() {
        System.out.println("Marca: " + marca);
        System.out.println("Modelo: " + modelo);
        System.out.println("Motor: " + motor);
        System.out.println("Combustible: " + combustible);
        System.out.println("Tipo: " + tipo);
        System.out.println("Puertas: " + puertas);
        System.out.println("Asientos: " + asientos);
        System.out.println("Velocidad Max: " + velocidadMax);
        System.out.println("Color: " + color);
        System.out.println("Velocidad Actual: " + velocidadActual);
        System.out.println("Automático: " + automatico); // ejercicio propuesto
        System.out.println("Multas: " + multas); // ejercicio propuesto
    }

    public static void main(String[] args) {
        Automovil a1 = new Automovil("Toyota", 2020, 2.0, Combustible.GASOLINA, TipoAuto.SUV, 4, 5, 180, Color.ROJO, true); // ejercicio propuesto
        a1.mostrarDatos();
        a1.velocidadActual = 100;
        System.out.println("Velocidad actual: " + a1.velocidadActual);
        a1.acelerar(20);
        System.out.println("Velocidad actual: " + a1.velocidadActual);
        a1.desacelerar(50);
        System.out.println("Velocidad actual: " + a1.velocidadActual);
        a1.frenar();
        System.out.println("Velocidad actual: " + a1.velocidadActual);
        System.out.println("Tiene multas: " + a1.tieneMultas()); // ejercicio propuesto
        System.out.println("Valor multas: " + a1.valorMultas()); // ejercicio propuesto
    }
}
